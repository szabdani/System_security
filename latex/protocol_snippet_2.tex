 % MACRO for whole diagram 
 % #1: total width of diagram 
 % #2: total height of diagram 
 % #3: nodes, paths, ... 
 \newcommand\protocolflow[3]{ 
 \begin{center}

 \end{tikzpicture} 
 \end{center} 
 }

 % #2: target coordinate 
 % #3: text above arrow 
 \newcommand{\package}[3]{ 
 \path[*-{latex[width=5pt, length=5pt]}] (#1) edge node [above] {#3} (#2); 
 }
